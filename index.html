const SHEET_NAME = '走行記録';
function doGet(e){if(e.parameter.action==='get')return getRecords();return ContentService.createTextOutput(JSON.stringify({error:'invalid'})).setMimeType(ContentService.MimeType.JSON);}
function doPost(e){const data=JSON.parse(e.postData.contents);if(data.action==='add')addRecord(data.record);return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON);}
function getSheet(){const ss=SpreadsheetApp.getActiveSpreadsheet();let sheet=ss.getSheetByName(SHEET_NAME);if(!sheet){sheet=ss.insertSheet(SHEET_NAME);sheet.appendRow(['ID','日付','車両','ドライバー','出庫(km)','帰社(km)','走行距離(km)','給油(L)','備考','記録日時']);sheet.setFrozenRows(1);}return sheet;}
function addRecord(r){getSheet().appendRow([r.id,r.date,r.vehicle,r.driver,r.start,r.end,r.km,r.fuel||'',r.note||'',new Date().toLocaleString('ja-JP')]);}
function getRecords(){const rows=getSheet().getDataRange().getValues();const records=rows.slice(1).map(row=>({id:row[0],date:row[1],vehicle:row[2],driver:row[3],start:row[4],end:row[5],km:row[6],fuel:row[7]||'',note:row[8]||''})).reverse();return ContentService.createTextOutput(JSON.stringify({records})).setMimeType(ContentService.MimeType.JSON);}
